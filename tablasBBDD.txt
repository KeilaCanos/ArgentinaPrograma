CREATE TABLE equipo(
nombreEquipo varchar(30) NOT NULL,
PRIMARY KEY (nombreEquipo))

CREATE TABLE fase(
ID_FASE INT NOT NULL,
PRIMARY KEY(ID_FASE))

CREATE TABLE partido(
ID_PARTIDO INT NOT NULL,
EQUIPO1 VARCHAR(30) NOT NULL,
GOLESEQ1 INT NOT NULL,
EQUIPO2 VARCHAR(30) NOT NULL,
GOLESEQ2 INT NOT NULL,
RONDA INT NOT NULL,
EQUIPO_GANADOR VARCHAR(30) NOT NULL,
FASE_CORRESP INT NOT NULL,
PRIMARY KEY (ID_PARTIDO),
FOREIGN KEY (RONDA) REFERENCES RONDA(ID_RONDA),
FOREIGN KEY (FASE_CORRESP) REFERENCES FASE(ID_FASE))

CREATE TABLE persona(
IDPERSONA INT NOT NULL,
NOMBREPER VARCHAR(30) NOT NULL,
PUNTOS INT NOT NULL,
PRIMARY KEY (IDPERSONA))

CREATE TABLE pronostico(
ID_PRONOSTICO INT PRIMARY KEY,
IDPERSONA INT NOT NULL,
ID_PARTIDO INT NOT NULL,
RESULTADO CHAR(1),
EQUIPO_RESULTADO VARCHAR(30) NOT NULL,
FOREIGN KEY (IDPERSONA) REFERENCES PERSONA(IDPERSONA),
FOREIGN KEY (ID_PARTIDO) REFERENCES PARTIDO(ID_PARTIDO))

CREATE TABLE ronda(
ID_RONDA INT PRIMARY KEY,
FK_FASE INT NOT NULL,
FOREIGN KEY (FK_FASE) REFERENCES FASE(ID_FASE))
